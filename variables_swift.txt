 /*
 Explícito, Tú le dices al compilador el tipo de dato
 (por ejemplo: String, Int, Double, Bool).
implícito: El compilador adivina el tipo de dato
 según el valor que le das. No se necesita escribir el tipo.
*/
 
 
 
 
// Variables explicitaS (Yo indicao el tipo)
 
var nombre: String = "Chilindrina"       // es un String
var edad: Int = 25                  // un número entero
var altura: Double = 1.75           //  número con decimales
var esEstudiante: Bool = true       // esto es verdadero o falso
var inicial: Character = "C"        // esto es un caracter unico
 
 
// Variables implicitas (la compu adivina el tipo)
 
var ciudad = "Roma"                // Como pusimos texto, sabe que es String
var precio = 10.50                 // Como hay decimales,  es Double
var cantidad = 100                 // número entero, es Int
var activo = false                 // Como es falso, indica un verdadero, por lo tanto booleano
var numeros = [1, 2, 3, 4, 5]        // Array de enteros (nuevo tipo), segun a mi ojo de buen cubero, aqui fue donde se repitio una 
 
 
 
// Constantes explicitas (yo indico el tipo y NO cambian. Se declaran con Let como de "dejala que sea constante". Si la vuelves a escribir debe de ser el mismo valor, si no, al programa le da amsiedad y se muere)
 
let pi: Double = 3.1416            // Número con decimales
let saludo: String = "Hola mundo"  // Texto fijo
let maxUsuarios: Int = 1000        // Número entero fijo
let esValido: Bool = true          // Verdadero o falso fijo
let simbolo: Character = "♥"       // Carácter único
 
 
 
//constantes implicitas (la maquina adivina el tipo y no se pueden cambiar)
 
let pais = "México"                // Texto, es String
let temperatura = 22.5             // Número con decimales, es Double
let diasSemana = 7                 // Número entero, es Int
let esMayorEdad = false            // Verdadero/falso, es Bool
let colores = ["Rojo", "Verde"]       // y aqui cambie de string a Array
